class Solution {
    public int leastInterval(char[] tasks, int n) {
        Map<Character, Integer> counts = new HashMap<>();
        int max = 0;
        for(char task : tasks){
            counts.put(task, counts.getOrDefault(task, 0) + 1);
            max = Math.max(counts.get(task), max);
        }
        int maxCount = 0;
        for(int val : counts.values()){
            if(val == max){
                maxCount++;
            }
        }
        
        int time = (n+1) * (max - 1) + maxCount;
        if(time < tasks.length){
            return tasks.length;
        }
        
        return time;
    }
}
